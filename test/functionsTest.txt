
struct Point{
  int x;
  int y;
};

func makePoint(int x, int y) Point {
  Point p = {x: x, y: y};
  return p;
} 

func sumPoint(Point n) int {
  return n.x + n.y;
}

func sumArr(array<int> arr) int {
  int sum = 0;
  int i = 0;
  while(i < len(arr)){
    sum = sum + arr[i];
    i = i + 1;
  }

  return sum;
}

func isPositive(int number) bool {
    if (number > 0) {
        return true;
    }
    return false;
}

func reverseString(string s) string {
    string result = "";
    int length = len(s);
    int i = length - 1;
    while (i >= 0) {
        result = result + s[i];
        i = i - 1;
    }
    return result;
}

func log(string a) void {
  print("LOG:", a)
}
