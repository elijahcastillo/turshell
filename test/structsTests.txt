
struct Friend {
  bool isClose;
  string name;
};

struct Person{
  int age;
  string name;
  Friend friend;
  array<float> money;
};

Person bob = {age: 22, name: "Bob", friend: {name: "Tim", isClose: true}, money: [521.34, 1087.81]};

print(bob.age)
print(bob.name)
print(bob.money)
print(bob.money[1])
print(bob.friend)
print(bob.friend.name)
