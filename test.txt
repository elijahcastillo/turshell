
struct Point{
  int x;
  int y;
};

func makePoint(int x, int y) Point {
  Point p = {x: x, y: y};
  return p;
} 


func sumPoint(Point n) int {
  return n.x + n.y;
}

func sumPoints(array<Point> arr) Point {
  Point sum = {x: 0, y:0};

  int i = 0;
  while(i < len(arr)){
    Point p = arr[i];
    sum.x = sum.x + p.x;
    sum.y = sum.y + p.y;
    i = i + 1;
  }

  return sum;
}



array<Point> po = [{x: 1, y:2}, {x: 6, y:4}, {x: 3, y: 3}];
Point sum = sumPoints(po);
print(sum)
